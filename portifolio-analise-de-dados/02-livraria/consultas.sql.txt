
-- 1. Livros com preço acima da média
SELECT titulo, preco FROM livros 
WHERE preco > (SELECT AVG(preco) FROM livros);

-- 2. Quantos livros cada autor tem
SELECT autor, COUNT(*) as quantidade 
FROM livros 
GROUP BY autor 
ORDER BY quantidade DESC;

-- 3. Categoria com maior quantidade de livros
SELECT categoria, COUNT(*) as total 
FROM livros 
GROUP BY categoria 
ORDER BY total DESC 
LIMIT 1;

-- 4. Qual categoria tem a maior média de preço?
SELECT categoria, AVG(preco) as media 
FROM livros 
GROUP BY categoria 
ORDER BY media DESC;
